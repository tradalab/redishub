# RedisHub

## Ubuntu 22 env

- golang: 
    - sudo apt install 
- nodejs: 
    - curl -fsSL https://deb.nodesource.com/setup_lts.x | sudo -E bash -
    - sudo apt install -y nodejs
- pnpm:
    - sudo corepack enable
- webview: 
    - sudo apt install -y libgtk-3-dev libwebkit2gtk-4.0-dev build-essential pkg-config
